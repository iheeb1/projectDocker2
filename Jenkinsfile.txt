pipeline {
    agent any
    
    environment {
        PYTHON_VERSION = '3.8'  
    }

    tools {
        // Define the Python tool with the desired version
        python "Python ${PYTHON_VERSION}"
    }

    stages {
        stage('Install Dependencies') {
            steps {
                script {
                    // Install required Python packages using pip
                    bat "python.exe -m pip install -r requirements.txt"
                }
            }
        }

        stage('Set Up Environment') {
            steps {
                script {
                    // Set up a virtual environment
                    bat "python -m venv venv"

                    // Activate the virtual environment
                    if (isUnix()) {
                        sh "source venv/bin/activate"
                    } else {
                        // On Windows, use `call` to activate the virtual environment
                        bat "call venv\\Scripts\\activate"
                    }
                }
            }
        }

        stage('Run Your Tests') {
            steps {
                script {
                    // Run your tests or other commands within the virtual environment
                    bat "python your_test_script.py"
                }
            }
        }
    }
}
