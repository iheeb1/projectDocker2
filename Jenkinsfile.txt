pipeline {
    agent any
    
    environment {
        PYTHON_VERSION = '3.8'  
    }

    stages { 
    

 stage('Build') {
       
         bat "docker-compose build"
   }
   stage("Test"){
        bat "py unittests.py"
        post {
                always {junit 'test-reports/*.xml'}
            }
    }

    stage("Run"){
        bat "docker-compose up "
    }
       
}
}
