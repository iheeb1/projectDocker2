pipeline {
    agent any
    
    environment {
        PYTHON_VERSION = '3.8'  
    }

    stages { 
    

 stage('Install Dependencies') {
            steps {
                script {
                    bat "python -m pip install -r requirements.txt"
                }
            }
        }
        stage('Set Up Environment') {
            steps {
                script {
                    bat "python${PYTHON_VERSION} -m venv venv"
                    bat "source venv/bin/activate"
                }
            }
        }
}
}
