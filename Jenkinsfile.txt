pipeline {
    agent any

    stages { 
    

 stage('Install Dependencies') {
            steps {
                script {
                    bat "py -m pip install -r App/requirements.txt"
                }
            }
        }
        stage('Set Up Environment') {
            steps {
                script {
                    // Set up a virtual environment
                    bat "py -m venv venv"
                    bat "source venv/bin/activate"
                }
            }
        }
}
}
