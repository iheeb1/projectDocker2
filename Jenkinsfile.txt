pipeline {
    agent any

    environment {
        PYTHON_VERSION = '3.8'  
    }

    stages {
        stage('Build') {
            steps {
                script {
                    sh "docker-compose build"
                }
            }
        }

        stage("Test") {
            steps {
                script {
                    sh "py unittests.py"
                }
            }
            post {
                always {
                    junit 'test-reports/*.xml'
                }
            }
        }

        stage("Run") {
            steps {
                script {
                    sh "docker-compose up"
                }
            }
        }
    }
}
