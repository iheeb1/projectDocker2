pipeline {
    agent any
    
    environment {
        PYTHON_VERSION = '3.8'  
    }

    stages {
        stage('Install Dependencies') {
            steps {
                script {
                    // Install required Python packages using pip
                    bat "python${PYTHON_VERSION} -m pip install -r requirements.txt"
                }
            }
        }

        stage('Set Up Environment') {
            steps {
                script {
                    // Set up a virtual environment
                    bat "python${PYTHON_VERSION} -m venv venv"

                    // Activate the virtual environment
                    if (isUnix()) {
                        bat "source venv/bin/activate"
                    } else {
                        bat "venv\\Scripts\\activate"
                    }
                }
            }
        }

        stage('Run Your Tests') {
            steps {
                script {
                    // Run your tests or other commands within the virtual environment
                    bat "python${PYTHON_VERSION} your_test_script.py"
                }
            }
        }
    }
}
